import React, { useEffect, useRef, useState } from 'react';
import ForceGraph3D from '3d-force-graph';
import * as THREE from 'three';
import data from './data.json';

const brandColors = {
  'EltaMD': '#f59e0b',
 // 'Neutrogena': '#3b82f6',
 // 'zoskinhealth': '#10b981',
 // 'Skinceuticals': '#8b5cf6',
 // 'Supergoop': '#ec4899'
};

const BrandThemeGraph3D = () => {
  const containerRef = useRef(null);
  const graphRef = useRef(null);
  const [stats, setStats] = useState({ reviews: 0, products: 0, themes: 0 });
  const [useEmbedding, setUseEmbedding] = useState(false);

  const generateEmbedding = (text, dimensions = 3) => {
    const hash = text.split('').reduce((acc, char) => {
      return ((acc << 5) - acc) + char.charCodeAt(0);
    }, 0);
    
    const coords = [];
    let seed = Math.abs(hash);
    for (let i = 0; i < dimensions; i++) {
      seed = (seed * 9301 + 49297) % 233280;
      coords.push((seed / 233280) * 400 - 200);
    }
    return coords;
  };

  // Create 3D text sprite with better sizing
  const createTextSprite = (text, color = '#ffffff', fontSize = 32, isBold = false) => {
    const canvas = document.createElement('canvas');
    const context = canvas.getContext('2d');
    canvas.width = 512;
    canvas.height = 64;
    
    context.fillStyle = color;
    context.font = `${isBold ? 'bold' : 'bold'} ${fontSize}px Arial`;
    context.textAlign = 'left';
    context.textBaseline = 'middle';
    context.fillText(text, 10, 32);
    
    const texture = new THREE.CanvasTexture(canvas);
    const spriteMaterial = new THREE.SpriteMaterial({ 
      map: texture,
      depthTest: false,
      depthWrite: false,
      transparent: true
    });
    const sprite = new THREE.Sprite(spriteMaterial);
    sprite.scale.set(18, 2.5, 1);
    sprite.renderOrder = 9999;
    
    return sprite;
  };

  // Create 3D sphere for legend
  const createLegendSphere = (color, size = 0.6) => {
    const geometry = new THREE.SphereGeometry(size * 2, 32, 32);
    const material = new THREE.MeshBasicMaterial({ 
      color,
      depthTest: false,
      depthWrite: false
    });
    const mesh = new THREE.Mesh(geometry, material);
    mesh.renderOrder = 10000;
    return mesh;
  };

  useEffect(() => {
    const maxRows = 2000;
    const sampleData = (Array.isArray(data) ? data : data.default).slice(0, maxRows);
    console.log('Loaded JSON records:', sampleData.length);

    const nodes = [];
    const links = [];
    const nodeMap = new Map();

    const addNode = (id, label, type, color, parent = null, sentiment = null, embeddingText = null) => {
      if (!nodeMap.has(id)) {
        const embedding = embeddingText ? generateEmbedding(embeddingText) : null;
        nodeMap.set(id, nodes.length);
        nodes.push({ 
          id, 
          label, 
          type, 
          color, 
          parent, 
          sentiment,
          embedding,
          fx: useEmbedding && embedding ? embedding[0] : undefined,
          fy: useEmbedding && embedding ? embedding[1] : undefined,
          fz: useEmbedding && embedding ? embedding[2] : undefined
        });
      }
      return nodeMap.get(id);
    };

    const themeColumns = [
      'Suits all skin types', 'Packaging Issues', 'Price Performance',
      'Customer Satisfaction', 'Age and Skin Benefits', 'Effectiveness',
      'Promotions', 'Convenience', 'Diverse Options'
    ];

    const subthemeMapping = {
      'Suits all skin types': 'Suits all skin types_Subtheme',
      'Packaging Issues': 'Packaging Issues_Subtheme',
      'Price Performance': 'Price Performance_Subtheme',
      'Customer Satisfaction': 'Customer Satisfaction_Subtheme',
      'Age and Skin Benefits': 'Age and Skin Benefits_Subtheme',
      'Effectiveness': 'Effectiveness_Subtheme',
      'Promotions': 'Promotions_Subtheme',
      'Convenience': 'Convenience_Subtheme',
      'Diverse Options': 'Diverse Options_Subtheme'
    };

    const uniqueProducts = new Set();
    const uniqueThemes = new Set();

    sampleData.forEach((row, idx) => {
      const brand = row.Brand || 'Unknown';
      const sku = row.SKU || 'Unknown SKU';
      const sentiment = row.Sentiment || 'neutral';
      const reviewText = row.Text || '';

      const brandColor = brandColors[brand] || '#6b7280';
      addNode(brand, brand, 'brand', brandColor, null, null, brand);

      const productId = `${brand}_${sku}`;
      uniqueProducts.add(sku);
      addNode(productId, sku, 'product', brandColor, brand, null, `${brand} ${sku}`);
      links.push({ source: brand, target: productId });

      themeColumns.forEach(theme => {
        if (row[theme]) {
          uniqueThemes.add(theme);
          const themeId = `${productId}_${theme}`;

          const themeValue = row[theme];
          let themeSentiment = 'neutral';
          if (themeValue.includes('positive')) themeSentiment = 'positive';
          else if (themeValue.includes('negative')) themeSentiment = 'negative';

          addNode(themeId, theme, 'theme', '#8b5cf6', productId, null, `${theme} ${reviewText.substring(0, 100)}`);
          links.push({ source: productId, target: themeId });

          const subthemeCol = subthemeMapping[theme];
          if (subthemeCol && row[subthemeCol]) {
            const subtheme = row[subthemeCol];
            const subthemeId = `${themeId}_${subtheme}_${idx}`;

            const sentimentColor =
              themeSentiment === 'positive' ? '#22c55e' :
              themeSentiment === 'negative' ? '#ef4444' : '#64748b';

            const subthemeLabel = `${subtheme} (${themeSentiment})`;
            addNode(subthemeId, subthemeLabel, 'subtheme', sentimentColor, themeId, themeSentiment, 
              `${subtheme} ${themeSentiment} ${reviewText.substring(0, 100)}`);
            links.push({ source: themeId, target: subthemeId });
          }
        }
      });

      const sentimentColor =
        sentiment === 'positive' ? '#22c55e' :
        sentiment === 'negative' ? '#ef4444' : '#64748b';
      const sentimentNodeId = `${productId}_sentiment_${sentiment}_${idx}`;
      addNode(sentimentNodeId, `Overall: ${sentiment}`, 'sentiment', sentimentColor, productId, sentiment, 
        `${sentiment} ${reviewText}`);
      links.push({ source: productId, target: sentimentNodeId });
    });

    setStats({
      reviews: sampleData.length,
      products: uniqueProducts.size,
      themes: uniqueThemes.size
    });

    const graphData = { nodes, links };

    const geometries = {
      brand: new THREE.SphereGeometry(12, 16, 16),
      product: new THREE.SphereGeometry(7, 16, 16),
      theme: new THREE.SphereGeometry(5, 16, 16),
      subtheme: new THREE.SphereGeometry(3.5, 16, 16),
      sentiment: new THREE.SphereGeometry(2.5, 16, 16)
    };

    const elem = containerRef.current;
    const Graph = ForceGraph3D()(elem)
      .graphData(graphData)
      .nodeAutoColorBy('type')
      .nodeLabel(node => {
        const matchedData = sampleData.find(record => {
          if (!node.parent) return false;
          const [brand, sku] = (node.parent || '').split('_');
          return record.Brand === brand && record.SKU === sku;
        });

        let html = `<div style="font-size:12px; line-height:1.4; color:#fff;">`;
        html += `<strong>${node.label}</strong><br>`;
        html += `<span style="opacity:0.8;">Type:</span> ${node.type}<br>`;

        if (node.parent) {
          const parentNode = nodes.find(n => n.id === node.parent);
          if (parentNode && parentNode.type === 'product') {
            html += `<span style="opacity:0.8;">Product:</span> ${parentNode.label}<br>`;
          }
        }

        if (matchedData) {
          if (matchedData.Text)
            html += `<span style="opacity:0.8;">Review Text:</span> ${matchedData.Text}<br>`;
          if (matchedData.Stars)
            html += `<span style="opacity:0.8;">Rating:</span> ${matchedData.Stars} stars<br>`;
          if (matchedData.Sentiment)
            html += `<span style="opacity:0.8;">Sentiment:</span> ${matchedData.Sentiment}<br>`;
        }

        if (node.parent) {
          const parentNode = nodes.find(n => n.id === node.parent);
          if (parentNode && parentNode.type === 'theme') {
            html += `<span style="opacity:0.8;">Parent:</span> ${parentNode.label}`;
          }
        }

        html += `</div>`;
        return html;
      })
      .nodeThreeObject(node => {
        const material = new THREE.MeshBasicMaterial({ color: node.color });
        const mesh = new THREE.Mesh(geometries[node.type], material);
        node.__threeObj = mesh;
        return mesh;
      })
      .linkColor(() => 'rgba(255,255,255,0.3)')
      .linkOpacity(0.3)
      .linkDirectionalParticles(0)
      .backgroundColor('#0f172a')
      .onNodeClick(node => {
        const info = `Node: ${node.label}\nType: ${node.type}${node.sentiment ? `\nSentiment: ${node.sentiment}` : ''}`;
        alert(info);
      });

    Graph.cameraPosition({ z: 300 });
    
    if (useEmbedding) {
      Graph.d3Force('charge').strength(-100);
      Graph.d3Force('link').distance(50);
    }
    
    graphRef.current = Graph;

    // Create 3D Legend Panel in Scene
    const scene = Graph.scene();
    const camera = Graph.camera();
    
    const legendGroup = new THREE.Group();
    legendGroup.name = 'legend';
    
    let yOffset = 20;
    const leftMargin = -22;
    const iconX = leftMargin + 10;
    const textX = leftMargin + 10;
    
    // Title
    const title = createTextSprite('LEGEND', '#ffffff', 48, true);
    title.position.set(leftMargin + 3, yOffset, 0.2);
    legendGroup.add(title);
    yOffset -= 6;
    
    // Node Types Section
    const nodeTypesTitle = createTextSprite('Node Types', '#f1f5f9', 45, true);
    nodeTypesTitle.position.set(leftMargin, yOffset, 0.2);
    legendGroup.add(nodeTypesTitle);
    yOffset -= 4;
    
    const nodeTypes = [
    //  { label: 'Brand', color: '#94a3b8', size: 1.5 },
     // { label: 'Product', color: '#94a3b8', size: 1.0 },
      { label: 'Theme', color: '#8b5cf6', size: 0.7 },
     // { label: 'Subtheme', color: '#94a3b8', size: 0.6 }
    ];
    
    nodeTypes.forEach(item => {
      const sphere = createLegendSphere(item.color, item.size);
      sphere.position.set(iconX, yOffset, 0.2);
      legendGroup.add(sphere);
      
      const label = createTextSprite(item.label, '#ffffff', 45);
      label.position.set(textX, yOffset, 0.2);
      legendGroup.add(label);
      yOffset -= 3.5;
    });
    
    yOffset -= 2;
    
    // Sentiment Section
    const sentimentTitle = createTextSprite('Sentiment', '#f1f5f9', 45, true);
    sentimentTitle.position.set(leftMargin, yOffset, 0.2);
    legendGroup.add(sentimentTitle);
    yOffset -= 4;
    
    const sentiments = [
      { label: 'Positive', color: '#22c55e' },
      { label: 'Negative', color: '#ef4444' },
      { label: 'Neutral', color: '#64748b' }
    ];
    
    sentiments.forEach(item => {
      const sphere = createLegendSphere(item.color, 0.7);
      sphere.position.set(iconX, yOffset, 0.2);
      legendGroup.add(sphere);
      
      const label = createTextSprite(item.label, '#ffffff', 45);
      label.position.set(textX, yOffset, 0.4);
      legendGroup.add(label);
      yOffset -= 3.5;
    });
    
    yOffset -= 2;
    
    // Brands Section
    const brandsTitle = createTextSprite('Brands', '#f1f5f9', 45, true);
    brandsTitle.position.set(leftMargin, yOffset, 0.2);
    legendGroup.add(brandsTitle);
    yOffset -= 4;
    
    Object.entries(brandColors).forEach(([brand, color]) => {
      const sphere = createLegendSphere(color, 0.7);
      sphere.position.set(iconX, yOffset, 0.2);
      legendGroup.add(sphere);
      
      const label = createTextSprite(brand, '#ffffff', 45);
      label.position.set(textX, yOffset, 0.2);
      legendGroup.add(label);
      yOffset -= 3.5;
    });
    
    scene.add(legendGroup);
    
    // Position legend in screen space (fixed to camera)
    const positionLegendInScreenSpace = () => {
      // Get camera frustum dimensions at a fixed distance
      const distance = 150;
      const vFov = camera.fov * Math.PI / 180;
      const height = 2 * Math.tan(vFov / 2) * distance;
      const width = height * camera.aspect;
      
      // Create a position in screen space (top-right corner)
      const screenPos = new THREE.Vector3(width * 0.38, height * 0.32, -distance);
      
      // Transform to world space relative to camera
      screenPos.applyQuaternion(camera.quaternion);
      screenPos.add(camera.position);
      
      legendGroup.position.copy(screenPos);
      legendGroup.quaternion.copy(camera.quaternion);
    };
    
    // Update on camera movement
    const animate = () => {
      positionLegendInScreenSpace();
      requestAnimationFrame(animate);
    };
    animate();
    
    console.log('Legend added to scene');

    return () => {
      if (graphRef.current) {
        graphRef.current._destructor();
        graphRef.current = null;
      }
      Object.values(geometries).forEach(geometry => {
        geometry.dispose();
      });
      nodes.forEach(node => {
        if (node.__threeObj && node.__threeObj.material) {
          node.__threeObj.material.dispose();
        }
      });
      
      // Clean up legend
      legendGroup.children.forEach(child => {
        if (child.geometry) child.geometry.dispose();
        if (child.material) {
          if (child.material.map) child.material.map.dispose();
          child.material.dispose();
        }
      });
      scene.remove(legendGroup);
      
      elem.innerHTML = '';
    };
  }, [useEmbedding]);

  return (
    <div className="w-full h-screen bg-slate-900 relative">
      <div ref={containerRef} className="w-full h-full" />
      
      <div className="absolute top-2 left-2 text-white p-2 bg-slate-800/70 rounded">
        <div>Reviews: {stats.reviews}</div>
        <div>Products: {stats.products}</div>
        <div>Themes: {stats.themes}</div>
        <div className="mt-2 pt-2 border-t border-slate-600">
          <label className="flex items-center gap-2 cursor-pointer">
            <input 
              type="checkbox" 
              checked={useEmbedding}
              onChange={(e) => setUseEmbedding(e.target.checked)}
              className="w-4 h-4"
            />
            <span className="text-sm">Use Embedding Layout</span>
          </label>
        </div>
      </div>
    </div>
  );
};

export default BrandThemeGraph3D;
